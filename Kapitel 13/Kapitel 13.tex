\subsection{Newtons gravititionslov}
	\begin{align}
		F_g=G\frac{mM}{r^2}
	\end{align}
	\kraft\\
	$G$: Gravitionskonstanten ($\frac{Nm^2}{kg^2}$)\\
	$m$: Massen af det lille legeme ($kg$)\\
	$M$: Massen af det store legeme ($kg$)\\
	$r$: Distancen mellem to legemer ($m$)

		\end{align}
